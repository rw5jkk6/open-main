```go
package main

import (
	"fmt"
	"strings"
	//"strings"
)

func main(){

   for i:=1; i <= 9; i++{
      slist := []string{}
      for j:=1; j<=9; j++{
         s := fmt.Sprintf("%2d*%2d=%2d", i, j, i*j)
         slist=append(slist, s)
      }
      s1 := strings.Join(slist, ",")
      fmt.Println(s1)
   }
}
```
